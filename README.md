import requests
import pytest

test_cases = [

    (2, 200, "Ervin Howell"),
    (999, 404, None)
]

@pytest.mark.parametrize("user_id,expected_status,expected_name", test_cases)

    assert response.status_code == expected_status
    if expected_name:
        assert response.json()["name"] == expected_name
